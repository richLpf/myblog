Gin搭建项目-打印日志

日志对于项目来说也是很重要的一部分，毕竟我们的程序是不可能没有bug的。如果日志打印在项目中，会比较零散，不方便管理，也不方便收集。这里记录下我的一些做法。

# 一、打印日志

打印日志主要在两个地方，请求参数和返回参数，这样出现问题，基本就可以排除90%的问题了。

func AddPassage(c *gin.Context) {
	var req model.ReqPassage
	if err := c.BindJSON(&req); err != nil {
		response.FailResponse(401, err.Error(), c)
		return
	}
	passage := model.Passage{
		ID:      req.ID,
		Title:   req.Title,
		Content: req.Content,
	}
	if err := services.NewPassage(passage); err != nil {
		response.FailResponse(401, err.Error(), c)
		return
	}
	response.SuccessResponse(nil, c)
}

我们调用一下，发现返回了bindJSON返回了EOF,这是因为gin，bindJSON方法只允许解析一次参数，第二次解析就会失败，解决办法如下

//RequestLogs 打印请求日志中间件
func RequestLogs() gin.HandlerFunc {
	return func(c *gin.Context) {
		method := c.Request.Method
		fmt.Println("header", c.Request)  //打印header信息
		if method == "POST" || method == "PUT" {
			buf := &bytes.Buffer{}  
			tea := io.TeeReader(c.Request.Body, buf)
			body, err := ioutil.ReadAll(tea)
			if err != nil {
				log.Panicf("read body err: %+v", err)
			}
			c.Request.Body = ioutil.NopCloser(buf)
			var params interface{}
			if err := binding.JSON.BindBody(body, &params); err != nil {
				fmt.Println("params err", params)
			}
			fmt.Println("request", params)
		}
		c.Next()
	}
}

# 二、定时清理访问日志

如果这样操作当然会出现其他问题，日志文件增加很快，服务器磁盘很快就满了，日志查询也很费劲

# 1、要增加告警

执行语句：查看系统内存使用情况

df -P | grep /dev | awk '{print $5}'
1

shell脚本：定时获取

#*************************************************************************
#  FileName     :               alarm.sh 
#*************************************************************************
#  Author       :               pengfei.lv
#  CreateDate   :               2020-05-16
#  Description  :               this script is mointoring the linux disk
#                               capacity, if disk used more than 90%, then notice
#*************************************************************************


#!/bin/bash
# 读取磁盘空间使用情况
for d in `df -P | grep /dev | awk '{print $5}' | sed 's/%//g'`
do
  if [ $d -gt 90 ]; then
    echo "内存使用大小$d"
    echo "当内容空间使用率大于90,告警通知,邮件，企业微信，短信。。。"
    exit 0;
  fi
done

# 2、每天自动切割日志,并清理7天以上日志

具体做法

（1）获取当前的日志

（2）将本来的日志按照时间命名复制一份

（3）将原来的日志清空

（4）查找7天以上的日志，删除

清空日志一般使用：cat /dev/null > file.log

xargs rm 传递删除命令给找到的文件

# /bin/bash

base_path='/var/lib/logs/'
echo $base_path

day=`date '+%Y%m%d'`
echo $day
# 先将本来的日志按时间命名复制一份
cp $base_path"file.log" $base_path"log"$day".log"
# 将原来的日志文件清空
cat /dev/null > $base_path"file.log"

\# 清理7天以上的日志
find /var/lib/logs/ -name '*.log' -and -mtime +7 -type f |xargs rm
